TITLE Programa Fibonacci
; Este programa calcula los primeros 7 valores de la secuencia de
; numeros Fibonacci usando un loop. La secuencia se describe por la siguiente
; formula Fib(1), Fib(2) = 1, Fib(n) = Fib(n - 1) + Fib(n - 2)
; Authores: Enrique Ariel Lopez Capistran, Javier Ledezma Guzman
INCLUDE    C:\Irvine\Irvine32.inc
INCLUDELIB C:\Irvine\Irvine32.lib

.data
PRIMERO DWORD ?
SEGUNDO DWORD ?
RESULTADO DWORD ?

.code
main PROC

    mov ecx, 12

    mov PRIMERO, 0
    mov SEGUNDO, 1

    Calcular:
            mov eax, PRIMERO
            add eax, SEGUNDO
            call DumpRegs
            mov RESULTADO, eax
            mov eax, SEGUNDO
            mov PRIMERO, eax
            mov eax, RESULTADO
            mov SEGUNDO, eax
    loop Calcular

main ENDP
END main

TITLE Programa: Copiar String Al Revez
; Este programa copia un string desde su fuente(source) hacia su objetivo final, para
; posteriormente ir volteando la cadena en el proceso.
; Autores: Enrique A. Lopez Capistran, Javier Ledezma Guzman
INCLUDE    C:\Irvine\Irvine32.inc
INCLUDELIB C:\Irvine\Irvine32.lib

.DATA
source	BYTE	"Fuente del String, profe pasenos porfas",0
target	BYTE	SIZEOF source DUP("#")

.CODE
main PROC
	mov	esi,OFFSET source + SIZEOF source - 2
	mov	edi,OFFSET target
	mov	ecx,SIZEOF source - 1
VoltearString:
	mov	al,[esi]
	mov	[edi],al
	inc	edi
	dec	esi
loop VoltearString
	mov	al,0
	mov	[edi],al
	
	call	Clrscr
	mov	edx,OFFSET source
	call	Writestring
	call	Crlf
	mov	edx,OFFSET target
	call	Writestring
	call	Crlf
main ENDP
END main
